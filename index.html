Q.4 Even Odd

    
        const findEvenOdd = (N, Arr) => {
            let even = 0;
            let odd = 0;
            for (let i = 0; i < N; i++) {
                if (Arr[i] % 2 === 0) {
                    even = even + Arr[i];
                } else {
                    odd = odd + Arr[i];
                    }
            }
            return [even, odd];
        };
    




    
Q.5 Find whether the number is present or not

    
        const Find_Num = (array,N,M) => {
            count = 0;
            for (i = 0; i < array.length; i++) {
                if (array[i] == M) {
                    count++;
                }
            }
            if (count >= 1) {
                return "YES";
            } else return "NO";

Q.10 Min and Max

    
        function arrayMin(arr) {
            min = 1000000;
            for (i = 0; i < arr.length; i++){
                if (arr[i] < min) {
                    min = arr[i];
                }
            }
            return min;
        }

        function arrayMax(arr) {
            max = 0;
            for (i = 0; i < arr.length; i++){
                if (arr[i] > max) {
                    max = arr[i];
                }
            }
            return max;
        }
    




    
Q.11 Birthday Game

    
        function Birthday_Game(arr, D, M) {
            let count = 0;
            let sum = 0;
            let n = arr.length;
            for (let i = 0; i < n - M + 1; i++) {
                for (let j = 0; j < M; j++) {
                sum += arr[i + j];
                }
                if (sum == D) {
                    count++;
                }
                sum = 0;
            }
            return count;
        }
